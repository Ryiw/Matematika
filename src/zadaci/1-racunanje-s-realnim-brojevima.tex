\begin{zadatak}

\begin{zadatak}

		\task jednaki

		\task parni
	\end{tasks}
\end{zadatak}

\begin{zadatak}

    \begin{tasks}(4)

		\task jednaki
		\task recipročni
		\task neparni
	\end{tasks}
\end{zadatak}
